#include <stdio.h>

int main()
{  
    int jogadores, partidas , N_marcou = 0, cont, j, i ; //N = jogadores M = partidas 

   printf("insira os jogadores e as partidas: ");
   scanf("%d", &jogadores);
   scanf("%d", &partidas);
   int gols [jogadores][partidas];

   for (i = 0; i < jogadores; i++){
    for (j = 0; j < partidas; j++){
        scanf("%d", &gols[i][j]);
    }
   }

   for ( i = 0; i < jogadores; i++){
    cont = 0;
    for (j = 0; j < partidas; j++){
        if(gols[i][j] == 0){
            cont++;
        }
    }
   }

   if (cont == 0){
    N_marcou++;
   }

   printf("%d", N_marcou);

   return 0;
}
